## API Request Example

```python
# Import necessary libraries
import pandas as pd
import requests

# Define function to create json for API
def df_to_dict(df):
    # Convert DataFrame to dictionary format
    return {f'P{i+1}': row.to_dict() for i, row in df.iterrows()}

# Import Data
df = pd.read_csv("")[['Datetime','Open','High','Low','Close']]

# Create JSON payload for API request
json_payload = df_to_dict(df)

# Specify Base URL
BASE_URL = "https://www.given-url.com"

# Set API endpoint URL
url = f"{BASE_URL}/genesis/ts-forecasting-model"

# Set headers for API request
headers = {"API-KEY":"123key456"}

# Send POST request to API
response = requests.post(url, json=json_payload, headers=headers)

# Parse JSON response
resp = response.json()

```
